# 环境搭建与编译

# 环境搭建

  • TypeScript的编译器是基于 Node.js 的，先安装 https://nodejs.org (opens new window)
node -v
  • 通过NPM包管理工具安装TypeScript编译器
npm i -g typescript
// 生成tsc命令，用于编译我们的TypeScript代码文件
tsc -v

# TypeScript编译

  • TypeScript代码
let str: string = '佐助'
  • 编译TypeScript代码
tsc <要编译的ts文件>
  • 编译选项
// 指定编译文件输出目录
tsc --outDir ./dist ./src/HelloTS.ts
// 指定编译的代码版本目标，默认为ES3
tsc --outDir ./dist --target ES6 ./src/HelloTS.ts
// 在监听模式下运行，文件发生改变时自动编译
tsc --outDir ./dist --target ES6 --watch ./src/HelloTS.ts
  • 编译配置文件
{
    "compilerOptions": {
        "outDir": "./dist",
        "target": "ES2015",
        "watch": true
    },
    "include": ["./src/**/*"] // 指定需要编译额ts文件目录，若未指定，则默认包含当前目录及子目录下的所有ts文件
}
  • 默认配置
// tsc默认会从当前目录开始去查找tsconfig.json文件，如果没找到，会逐级向上搜索父目录
tsc
// 指定配置文件，--project/-p
tsc -p ./c.json
// 指定配置文件目录，-p
tsc -p ./src

# 基础类型

let name: string = '法外狂徒张三'
let num: numer = 100;

name = 100; // error

# 空和未定义类型

  • Null和Undefined这两种类型有且只有一个值，在标注一个变量为Null和Undefined类型，那就表示该变量不能修改了
  • 默认情况下Null和Undefined是所有类型的子类型，所以可以把Null和Undefined赋值给其他类型的变量
let un: undefined;
un = 1; // error
let nul: null;
nul = 1; // error
let a: string = '奥利给'
a = undefined; // ok
a = null; // ok
  • 小技巧：指定strictNullChecks配置为true，可以有效的检测null值数据，避免很多常见问题，建议对可能出现的null和undefined进行容错处理，使程序更加严谨
let ele = $('.box');
if(ele) {
    ele.style.display = 'none';
}

# 对象类型

  • 字面量
let obj: {x: number, y: string} = {
    x: 1,
    y: 'giao'
}
  • 内置对象类型
let d: Date = new Date();
let set: Set = new Set();
  • 包装类型，如String、Number、Boolean，特别关注String类型和string类型的区别
// String对象不可以赋值给string类型
let a: string;
a = '1';
a = new String('1'); // error
// 可以将string类型赋值给String对象
let b: String;
b = new String('1');
b = '1'; // ok

# 数组类型

// 标注
let arr1: string[] = [];
arr1.push('鸣人'); // ok
arr1.push(1); // error
// 泛型标注
let arr2: Array<number> = [];
arr2.push(100); // ok
arr2.push('卡卡西'); // error

# 元组类型

  • 元组类似数组，但是存储的元素类型不必相同

但需注意

  • 初始化数据的个数以及对应位置标注类型必须一致
  • 越界数据必须是元组标注中的类型之一（标注越界数据可以不用对应顺序-联合类型）
  • 如果未开启strictNullChecks: true，则对应位会使用undefined进行初始化
let data1: [string, number] = ['动感超人', 100];
data1.push(100); // ok
data1.push('100'); // ok
data1.push(true); // error

# 枚举类型

介绍

枚举的作用是组织收集一组关联数据的方式 enum 枚举名称 { key1 = value1, key2 = value2 }

  • key不能是数字
  • value可以是数字，称为数字类型枚举，也可以是字符串，称为字符串类型枚举，但不能是其他值，默认数字是0
  • 第一个枚举值或者前一个枚举值为数字时，可以省略赋值，其值为 前一个数字+1
  • 数字类型枚举
enum HTTP_CODE {
    OK = 200,
    NOT_FOUND = 404
};
HTTP_CODE.OK // 200
  • 字符串类型枚举
enum URLS {
    USER_REGISETER = '/user/register',
    NOT_FOUND = 404 '/user/login'
};

# 无值类型

和Never类型区分

void时执行了return，只是没有值，never是不会执行return，比如抛出错误，导致函数终止执行

function fn(): void {
    // 无值
}

# Never类型

注意

  • 当一个函数永远不可能执行return的时候，返回就是never
  • never类型是所有其他类型的子类
  • 其他不能赋值给never，即使是any
function fn(): never {
    throw new Error('error');
}
let n: never;
let str: string;
str = n; // ok
n = str; // error

# 任意类型

有时并不确定这个值是什么类型或者不需要对该值进行类型检测，就可以标注为any类型

  • 任何值都可以赋值给any类型
  • any类型也可以赋值给任意类型
  • any有任意属性和方法
  • 小技巧，指定nolmplicitAny为true，当函数参数出现隐含的any类型时报错
function fn(): never {
    throw new Error('error');
}
let n: never;
let str: string;
str = n; // ok
n = str; // error

# 未知类型

unknow 3.0版本中新增，属于安全版的any，但是与any不同的是

  • unknow仅能赋值给unknown、any
  • unknow没有任何属性和方法
