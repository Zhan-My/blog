# 类型深入

# 联合类型

function css(ele: Element, attr: string, value: string|number) {}
let box = $('.box');
if(box) {
    css(box, 'width', '100px');
    css(box, 'opacity', 0.5);
    css(box, 'opacity', [1,2]);
}

# 交叉类型

  • 例如给对象进行扩展
interface o1 {x: number, y: string};
interface o2 {z: number};
let o: o1&o2 = Object.assign({}, {x: 1, y: '2'}, {z: 3});

# 字面量类型

  • 有时候希望标注的不是某个类型，而是一个固定值，就适用字面量类型
function css(ele: Element, value: '专业团队'|'黑人'|'巴啦啦小魔仙') {}
let box = $('.box');
if(box) {
    css(box, '专业团队'); // ok
    css(box, '黑人'); // ok
    css(box, '法外狂徒张三'); // error
}

# 类型别名

  • 有时候类型标注比较复杂，可以给标注起个简单的名字
type val = '专业团队'|'黑人'|'巴啦啦小魔仙'
function css(ele: Element, value: val) {}
let box = $('.box');
if(box) {
    css(box, '法外狂徒张三'); // error
}

# 类型推导

  • TS编译器会根据当前上下文主动的推导出对应的类型标注，这个过程发生在：
  • 初始化变量
let a = 1;
a = 'a'; // 不能将类型“"a"”分配给类型“number”
  • 设置函数默认参数值
function fn(x = 1) {
    x.substring(1, 3); // 不能将类型“"a"”分配给类型“number”
}
  • 返回函数值
function add(x: number, y: number): number {
    return x + y;
}
// 同上，function add(x: number, y: number): number
function add(x: number, y: number) {
    return x + y;
}

# 类型断言

  • 有时候需要标注一个更加精确的类型（比如img从Element类型缩小至img类型），以便能够使用精确类型的属性（比如src）
// Element
let img = $('#img');
// img
let img = <HEMLImageElement>$('#img');
// 或者
let img = $('#img') as HEMLImageElement;

注意

断言只是一种预判，并不会使数据本身产生实际的作用

let a = 1;
let b = <String>a; // error

# 类型操作符

# typeof

  • 获取值的类型，抽取类型，进行复用，注：typeof操作的是值
let colors = {
    color1: 'red',
    color2: 'blue'
}
let t = typeof colors; // "string" | "number" | "bigint" | "boolean" | "symbol" | "undefined" | "object" | "function"
type tColors = typeof colors;
let newColors: tColors;

# keyof

  • 获取类型所对应key的集合，返回值是key的联合类型，注：keyof操作的是类型
interface Person {
    name: string;
    age: number;
}
type personKeys = keyof Person; // 'name'|'age'
function getPersonVal(k: personKeys) {
    return p1[k];
}
getPersonVal('gender'); // error
  • 从一个变量中获取类型
let p2 = {
    name: '张三',
    age: 18,
    gender: '男'
}
function getPersonVal(k: keyof typeof p2) {
    return p2[k]
}
getPersonVal('gender');
  • keyof的一个经典作用
// 如果想要正常只能在给attr加联合类型
function css(ele: Element, attr: 'width'|'opacity') {
    return getComputedStyle(ele)[attr];
}
let box = document.querySelector('.box');
if(box) {
    css(box, 'opacity'); // ok
    css(box, 'height'); // error
}
// 想要正常导入所有attr属性，就可以进入vscode编辑器自带的lib.dom.d.ts库去查找接口
function css(ele: Element, attr: keyof CSSStyleDeclaration) {
    return getComputedStyle(ele)[attr];
}
let box = document.querySelector('.box');
if(box) {
    css(box, 'opacity'); // ok
    css(box, 'height'); // ok
}

# in

  • 针对值进行操作，用来判断值中是否包含指定的key
let n = 'name' in {name: 'xxx', age: 18}; // true
let g = 'gender' in {name: 'xxx', age: 18}; // false
  • 针对类型进行操作的话，内部使用for...in对类型进行遍历
interface Person {
    name: string;
    age: number;
}
type personKeys = keyof Person;
type newPerson = {
    [k in personKeys]: string;
    [k in 'name'|2|symbol|object]: string; // error! in后面的类型值必须是string或number或symbol
}

# extends

interface Person {
    name: string;
    age: number;
}
interface Person2 extends Person {
    gender: string;
}
let p: Person2 = {
    name: '张三',
    age: 18,
    gender: '男'
}

# 保护类型

  • 使用断言（不推荐）
if((<string>str).toUpperCase) {
    return (<string>str).toUpperCase();
} else {
    (<string[]>str).push();
}
  • 使用typeof，ts能够把typeof识别为类型保护，作为类型检查依据，不仅仅是在if中有效，在else中也是有效的
function toUpperCase(str: string|string[]) {
    if(typeof str === 'string') {
        str.toUpperCase();
    } else {
        str.push();
    }
}
  • 使用instanceof，ts也能够把instanceof识别为类型保护，针对细化的对象类型判断可以使用它来处理
function toUpperCase(str: string|string[]) {
    if(instanceof str === 'string') {
        str.toUpperCase();
    } else {
        str.push();
    }
}
  • 自定义类型保护
function canEach(data: Element[]|NodeList|Element): data is Element[]|NodeList {
    return (<NodeList>data).forEach !== undefined;
}
function fn(elements: Element[]|NodeList|Element) {
    if(canEach(elements)) {
        elements.forEach(() => {});
    } else {
        elements.classList.add('box');
    }
}
