    # 节流

    节流

    所谓节流，就是指连续触发事件但是在 n 秒中只执行一次函数

    # 时间戳版

    function throttle(fn, wait) {
    let previous = 0;
    return function(...args) {
        let now = Date.now();
        if (now - previous > wait) {
            fn.apply(this, args);
            previous = now;
        }
    }
}

    # 定时器版

    function throttle(fn, wait) {
    let timeout;
    return function(...args) {
        if (!timeout) {
            timeout = setTimeout(() => {
                timeout = null;
                fn.apply(this, args)
            }, wait)
        }

    }
}

    # 二者结合版

    /**
 * @desc 函数节流
 * @param fn 函数
 * @param wait 延迟执行毫秒数
 * @param type 1 表时间戳版，2 表定时器版
 */
function throttle(fn, wait ,type) {
    if(type===1){
        let previous = 0;
    }else if(type===2){
        let timeout;
    }
    return function(...args) {
        if(type===1){
            let now = Date.now();
            if (now - previous > wait) {
                fn.apply(this, args);
                previous = now;
            }
        }else if(type===2){
            if (!timeout) {
                timeout = setTimeout(() => {
                    timeout = null;
                    fn.apply(this, args)
                }, wait)
            }
        }

    }
}
