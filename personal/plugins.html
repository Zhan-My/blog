# 下拉刷新&上拉分页加载（vant）

<template>
  <div class="hello">
    <van-pull-refresh v-model="refreshing" @refresh="onRefresh">
      <van-list
        v-model="loading"
        :error.sync="error"
        error-text="请求失败，点击重新加载"
        :finished="finished"
        :finished-text="list.length > 0 ? '人类底线': '毛也没有'"
        @load="onLoad"
      >
        <van-cell v-for="(item, index) in list" :key="index" :title="item.text" />
      </van-list>
    </van-pull-refresh>
  </div>
</template>

<script>
import Vue from 'vue';
import { List, Cell, PullRefresh } from 'vant';
Vue.use(List).use(Cell).use(PullRefresh)
export default {
  name: 'van-list',
  data() {
    return {
      list: [],
      error: false,
      loading: false,
      finished: false,
      refreshing: false,
      page: 0,
      count: 20
    };
  },
  methods: {
    onLoad() {
      this.page += 1
      this.getList()
    },
    getList() {
      // 异步更新数据
      setTimeout(() => {
        if (this.refreshing) {
          this.page = 1;
          this.list = [];
          this.refreshing = false;
        }
        this.$http({
          methods: 'get',
          url: 'https://api.apiopen.top/getJoke?page=' + this.page + '&count=' + this.count + '&type=image'
        }).then(res => {
          let { code, result } = res.data
          if(code === 200) {
            this.list = [...this.list, ...result]
            this.$nextTick(() => {
              this.loading = false
              if (this.list.length >= 100) { // 一般情况下这里后台返回总条数，作为判断条件
                this.finished = true
              }
            })
          }
        }).catch(() => {
          this.error = true
        });
      }, 1000);
    },
    onRefresh() {
      this.loading = true;
      this.finished = false;
      this.onLoad();
    },
  }
}
</script>

# 子组件

<template>
    <div>
      <input :value="value" @input="onInput" />
    </div>
</template>

<script>
export default {
    model: { // 如果不使用model属性，$emit('input', xxx)也是一样的
      prop: 'value',
      event: 'change'
    },
    props: {
      value: {
        type: String,
        default() {
          return ''
        }
      }
    },
    methods: {
      onInput(e) {
        this.$emit('change', e.target.value)
      }
    }
};
</script>

# 水印

watermark({
  url = "",
  textAlign = "left",
  textBaseline = "bottom",
  font = "48px PingFangSC-Medium",
  fillStyle = "#FF3B30",
  content = "",
  cb = null
} = {}) {
  const img = new Image();
  img.src = url;
  img.crossOrigin = "anonymous";
  img.onload = function () {
    const canvas = document.createElement("canvas");
    const _ix = img.width;
    const _iy = img.height;
    canvas.width = 500;
    canvas.height = _iy*(500/_ix);
    const ctx = canvas.getContext("2d");
    ctx.drawImage(img, 0, 0);
    ctx.textAlign = textAlign;
    ctx.textBaseline = textBaseline;
    ctx.font = font;
    ctx.fillStyle = fillStyle;
    const time = content.split('$')[0];
    const addr = content.split('$')[1];
    ctx.fillText(time, _ix * 0.05, _iy * 0.80, _ix); // 水印时间
    ctx.fillText(addr, _ix * 0.05, _iy * 0.90, _ix); // 水印地址
    const base64Url = canvas.toDataURL();
    cb && cb(base64Url);
  };
},

# 获取文件后缀

getFileType(filePath) {
  let startIndex = filePath.lastIndexOf(".");
  if (startIndex != -1)
    return filePath
      .substring(startIndex + 1, filePath.length)
      .toLowerCase();
  else return "";
},
